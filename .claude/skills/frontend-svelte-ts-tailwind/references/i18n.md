# i18n — Paraglide JS v2

## Setup

```bash
npm install @inlang/paraglide-js
```

```typescript
// vite.config.ts
import { paraglideVitePlugin } from '@inlang/paraglide-js';

export default defineConfig({
  plugins: [
    paraglideVitePlugin({
      project: './project.inlang',
      outdir: './src/lib/paraglide',
      strategy: ['cookie', 'globalVariable', 'baseLocale']
    }),
    tailwindcss(),
    sveltekit(),
  ]
});
```

## Locale Helpers

```typescript
// src/lib/i18n/paraglide.ts
import { isLocale, setLocale } from '$lib/paraglide/runtime';

export const SUPPORTED_LOCALES = ['en', 'zh-TW', 'zh-CN'] as const;
export type Locale = (typeof SUPPORTED_LOCALES)[number];

export function setLangCookie(locale: string): void {
  if (!isLocale(locale) || typeof document === 'undefined') return;
  const secure = location.protocol === 'https:' ? '; Secure' : '';
  document.cookie = `lang=${locale}; path=/; max-age=31536000; SameSite=Lax${secure}`;
}

export function getLangCookie(): Locale | null {
  if (typeof document === 'undefined') return null;
  const m = document.cookie.match(/(?:^|;\s*)lang=([^;]+)/);
  const v = m?.[1] ?? null;
  return v && isLocale(v) ? (v as Locale) : null;
}

export function detectLocale(): Locale {
  const cookie = getLangCookie();
  if (cookie) return cookie;
  if (typeof navigator === 'undefined') return 'en';
  const lang = navigator.language.toLowerCase();
  if (/^zh-(hant|hk|mo|tw)/i.test(lang)) return 'zh-TW';
  if (/^zh-(hans|cn|sg)/i.test(lang)) return 'zh-CN';
  return 'en';
}

export function switchLocale(locale: Locale): void {
  setLocale(locale);
  setLangCookie(locale);
}
```

## Component Usage

```svelte
<script lang="ts">
  import * as m from '$lib/paraglide/messages';
  import { getLocale } from '$lib/paraglide/runtime';
  import { switchLocale, SUPPORTED_LOCALES } from '$lib/i18n/paraglide';

  let currentLocale = $state(getLocale());
</script>

<h1>{m['dashboard.title']()}</h1>

<!-- Language switcher -->
<div class="flex gap-2">
  {#each SUPPORTED_LOCALES as locale}
    <button
      onclick={() => { switchLocale(locale); currentLocale = locale; }}
      class={cn('px-3 py-1 rounded text-sm', currentLocale === locale && 'bg-primary text-white')}
    >
      {locale}
    </button>
  {/each}
</div>
```

## Root Layout Locale Init

```typescript
// src/routes/+layout.svelte
import { onMount } from 'svelte';
import { detectLocale, switchLocale } from '$lib/i18n/paraglide';

onMount(() => {
  const locale = detectLocale();
  switchLocale(locale);
});
```
